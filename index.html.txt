        window.updatePresence = async function(id, presence) {
            const student = allStudents.find(s => s.id === id);
            if (!student) return;
            if (presence === "PRESENT") student.motif = "";
            await studentsCollection.doc(id).update({ presence, motif: student.motif, lastModified: firebase.firestore.FieldValue.serverTimestamp(), modifiedBy: deviceId });
        };
        window.updateMotif = async function(id, motifRaw) {
            const student = allStudents.find(s => s.id === id);
            if (!student) return;
            let finalMotif = motifRaw;
            if (motifRaw === "__AUTRE__") {
                const newMotif = prompt("✏️ Motif d'absence :");
                if (!newMotif || !newMotif.trim()) return;
                finalMotif = newMotif.trim().toUpperCase();
                if (!customMotifs.includes(finalMotif)) {
                    customMotifs.push(finalMotif);
                    await settingsDocRef.update({ customMotifs });
                }
            }
            student.motif = finalMotif || "";
            student.presence = "ABSENT";
            await studentsCollection.doc(id).update({ motif: student.motif, presence: "ABSENT", lastModified: firebase.firestore.FieldValue.serverTimestamp(), modifiedBy: deviceId });
        };
        window.editStudent = async function(id) {
            const student = allStudents.find(s => s.id === id);
            if (!student) return;
            const newNom = prompt("Nouveau nom :", student.nom);
            if (newNom && newNom.trim()) student.nom = newNom.trim().toUpperCase();
            const newPrenom = prompt("Nouveau prénom :", student.prenom);
            if (newPrenom && newPrenom.trim()) student.prenom = newPrenom.trim().toUpperCase();
            await studentsCollection.doc(id).update({ nom: student.nom, prenom: student.prenom, lastModified: firebase.firestore.FieldValue.serverTimestamp(), modifiedBy: deviceId });
        };
        window.deleteStudent = async function(id) {
            if (!confirm("Supprimer définitivement cet élève ?")) return;
            await studentsCollection.doc(id).delete();
        };

        function startSync() {
            studentsCollection.onSnapshot(snapshot => {
                allStudents = [];
                snapshot.forEach(doc => allStudents.push(doc.data()));
                allStudents.sort((a,b) => a.nom.localeCompare(b.nom, 'fr', {sensitivity:'base'}) || a.prenom.localeCompare(b.prenom, 'fr'));
                renderTable();
                syncIndicator.innerHTML = "✅ Synchronisé";
                syncIndicator.style.color = "#15803d";
            }, error => {
                console.error("Firestore onSnapshot error:", error);
                syncIndicator.innerHTML = "⚠️ Erreur de connexion Firestore";
                syncIndicator.style.color = "#b91c1c";
            });
            settingsDocRef.onSnapshot(doc => {
                if (doc.exists) {
                    const data = doc.data();
                    customMotifs = data.customMotifs || [];
                    if (data.lieu) lieuInput.value = data.lieu;
                    if (data.date) dateInput.value = data.date;
                } else {
                    settingsDocRef.set({ customMotifs: [], lieu: "BENSLIMANE", date: "01/05/2026" });
                }
                renderTable();
            });
        }
